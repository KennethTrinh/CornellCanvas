% Draw the Face
x0= [0 2 2 0 0 nan 0.25 0.75 nan 1.25 1.75 nan 0.5 0.5 1.5 1.5];
y0= [0 0 2 2 0 nan 1.5 1.75 nan 1.75 1.5 nan 0 1 1 0];
r0 = [x0;y0];

n = 100;
t = linspace(0,2*pi,n);
Fig = figure;

filename = 'Sad face.gif';
for i=1:1:n
    drop = t(i); % The motoin function
    R = [ cos(t(i))  -sin(t(i)); % Rotation function
      sin(t(i))  cos(t(i))];
    r = R* r0;
    r = bsxfun(@minus, r, [0;drop]);
    x=r(1,:);
    y=r(2,:);
    plot(x,y)
    axis('equal'); 
axis([-8 8 -8 8]);

% Get Gif
drawnow 
frame = getframe(Fig); 
    im = frame2im(frame); 
    [imind,cm] = rgb2ind(im,256);
    if i == 1
        imwrite(imind,cm,filename,'gif','WriteMode','overwrite', 'Loopcount',inf);
    else
        imwrite(imind,cm,filename,'gif','WriteMode','append','DelayTime',0.02);
    end
end
