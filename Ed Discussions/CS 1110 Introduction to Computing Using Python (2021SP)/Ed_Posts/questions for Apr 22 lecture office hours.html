class Person():
    """
    An instance is a Person with a name and possibly two parents.

    Examples:
    To create a Person with the name 'Pat' and no parents, type:
        p1= person.Person('Pat', None, None)
    To create a Person with the name 'Kai' and p1 (a Person) as one parent, type:
        p2= person.Person('Kai', p1, None)
    or
        p2= person.Person('Kai', None, p1)
    """

    def __init__(self, name, parent1, parent2):
        """
        Creates a new Person with the given name and parents.

        Preconditions: name is a non-empty string. parent1 and parent2 are each
        a Person or None.  parent1 and parent2 are distinct if they are not None.
        """
        self.name= name
        self.parent1= parent1
        self.parent2= parent2
