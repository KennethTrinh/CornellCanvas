        for tc in test_cases:
            inputlist = list(tc)
            if verbose:
                print('\ttrying ' + str(inputlist))
            ca.assert_equals(test_cases[tc], reverse_fn(inputlist))

        # check for returning a copy
        mylist = [3]
        ca.assert_equals(False, mylist is reverse_fn(mylist))
